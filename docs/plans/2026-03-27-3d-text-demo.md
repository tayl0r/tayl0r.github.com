# 3D Text Demo Page Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Create a standalone demo page at `/3d-text-demo/` showcasing 15 variations of 3D text ("Hello Claude") using 5 techniques across 3 visual themes with 3 animation styles.

**Architecture:** A Vite multi-page app entry at `3d-text-demo/index.html`. Each section (Dark/Moody static, Clean/Minimal whole-word, Playful/Bold per-letter) shows all 5 techniques side-by-side. Canvas-based demos (Troika, TextGeometry) share a scene utility and animation loop. DOM-based demos (CSS, ztext, GSAP) live in HTML + lightweight JS init.

**Tech Stack:** Vite, TypeScript, Three.js, troika-three-text, ztext, gsap (SplitText), CSS transforms/text-stroke

**5 Techniques:**
1. **Troika-three-text** — SDF text in Three.js (canvas)
2. **CSS perspective + text-stroke** — pure CSS, no JS (DOM)
3. **Three.js TextGeometry** — extruded 3D mesh letters (canvas)
4. **ztext.js** — DOM layer stacking for fake depth (DOM)
5. **GSAP SplitText** — per-char splitting + 3D CSS animation (DOM)

### Task 3: Create shared scene utility

**Files:**
- Create: `src/demo/create-scene.ts`

Shared factory for canvas-based demos (Troika + TextGeometry).

```ts
import * as THREE from "three";

export interface DemoScene {
	scene: THREE.Scene;
	camera: THREE.PerspectiveCamera;
	renderer: THREE.WebGLRenderer;
}

export function createDemoScene(canvas: HTMLCanvasElement): DemoScene {
	const container = canvas.parentElement as HTMLElement;
	const width = container.clientWidth;
	const height = container.clientHeight;

	const scene = new THREE.Scene();

	const camera = new THREE.PerspectiveCamera(50, width / height, 0.1, 100);
	camera.position.z = 5;

	const renderer = new THREE.WebGLRenderer({
		canvas,
		antialias: true,
		alpha: true,
	});
	renderer.setSize(width, height);
	renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));

	window.addEventListener("resize", () => {
		const w = container.clientWidth;
		const h = container.clientHeight;
		camera.aspect = w / h;
		camera.updateProjectionMatrix();
		renderer.setSize(w, h);
	});

	return { scene, camera, renderer };
}
```

**`src/demo/troika-playful.ts`** — Per-letter bobbing + rotation:

```ts
import { Text } from "troika-three-text";
import { createDemoScene } from "./create-scene";
import type { Demo } from "./troika-dark";

const COLORS = [
	0xff3366, 0xff6633, 0xffcc33, 0x33cc66, 0x3399ff,
	0x9933ff, 0xff33cc, 0xff3366, 0x33ccff, 0x66ff33, 0xffcc33,
];

interface LetterObj {
	text: Text;
	baseY: number;
	index: number;
}

function layoutWord(word: string, y: number, startIdx: number, fontSize: number): LetterObj[] {
	const spacing = fontSize * 0.65;
	const totalWidth = (word.length - 1) * spacing;
	const letters: LetterObj[] = [];

	for (let i = 0; i < word.length; i++) {
		const t = new Text();
		t.text = word[i];
		t.fontSize = fontSize;
		t.anchorX = "center";
		t.anchorY = "middle";
		t.color = COLORS[(startIdx + i) % COLORS.length];
		t.outlineWidth = 0.008;
		t.outlineColor = COLORS[(startIdx + i) % COLORS.length];
		t.outlineBlur = 0.02;
		t.position.set(-totalWidth / 2 + i * spacing, y, 0);
		t.sync();
		letters.push({ text: t, baseY: y, index: startIdx + i });
	}
	return letters;
}

export function initPlayfulDemo(canvas: HTMLCanvasElement): Demo {
	const { scene, camera, renderer } = createDemoScene(canvas);

	const allLetters = [
		...layoutWord("Hello", 0.5, 0, 0.7),
		...layoutWord("Claude", -0.5, 5, 0.7),
	];

	for (const l of allLetters) scene.add(l.text);

	return {
		render: (time: number) => {
			for (const l of allLetters) {
				const phase = l.index * 0.4;
				l.text.position.y = l.baseY + Math.sin(time * 1.5 + phase) * 0.15;
				l.text.rotation.y = Math.sin(time * 1.2 + phase) * 0.4;
			}
			renderer.render(scene, camera);
		},
	};
}
```

```ts
		fontPromise = new Promise((resolve, reject) => {
			new FontLoader().load(
				"https://cdn.jsdelivr.net/npm/three@0.183.2/examples/fonts/helvetiker_bold.typeface.json",
				resolve,
				undefined,
				reject,
			);
		});
	}
	return fontPromise;
}

export function createTextMesh(
	char: string,
	font: Font,
	size: number,
	depth: number,
	material: THREE.Material,
): THREE.Mesh {
	const geometry = new TextGeometry(char, {
		font,
		size,
		depth,
		curveSegments: 12,
		bevelEnabled: true,
		bevelThickness: 0.03,
		bevelSize: 0.02,
		bevelSegments: 5,
	});
	geometry.computeBoundingBox();
	if (geometry.boundingBox) {
		const cx = (geometry.boundingBox.max.x - geometry.boundingBox.min.x) / 2 + geometry.boundingBox.min.x;
		const cy = (geometry.boundingBox.max.y - geometry.boundingBox.min.y) / 2 + geometry.boundingBox.min.y;
		geometry.translate(-cx, -cy, 0);
	}
	return new THREE.Mesh(geometry, material);
}

export function layoutTextLine(
	text: string,
	font: Font,
	size: number,
	depth: number,
	material: THREE.Material,
	y: number,
	spacing: number,
): THREE.Mesh[] {
	const meshes: THREE.Mesh[] = [];
	const totalWidth = (text.length - 1) * spacing;

	for (let i = 0; i < text.length; i++) {
		const mesh = createTextMesh(text[i], font, size, depth, material);
		mesh.position.set(-totalWidth / 2 + i * spacing, y, 0);
		meshes.push(mesh);
	}
	return meshes;
}
```

```ts
import * as THREE from "three";
import { createDemoScene } from "./create-scene";
import { createTextMesh, loadFont } from "./textgeo-shared";
import type { Demo } from "./troika-dark";

const COLORS = [
	0xff3366, 0xff6633, 0xffcc33, 0x33cc66, 0x3399ff,
	0x9933ff, 0xff33cc, 0xff3366, 0x33ccff, 0x66ff33, 0xffcc33,
];

interface AnimatedLetter {
	mesh: THREE.Mesh;
	baseY: number;
	index: number;
}

export async function initTextGeoPlayful(canvas: HTMLCanvasElement): Promise<Demo> {
	const { scene, camera, renderer } = createDemoScene(canvas);

	const ambient = new THREE.AmbientLight(0xffffff, 0.6);
	scene.add(ambient);
	const dir = new THREE.DirectionalLight(0xffffff, 0.8);
	dir.position.set(5, 5, 10);
	scene.add(dir);

	const font = await loadFont();
	const allLetters: AnimatedLetter[] = [];
	const size = 0.6;
	const spacing = 0.85;

	function addWord(word: string, y: number, startIdx: number) {
		const totalWidth = (word.length - 1) * spacing;
		for (let i = 0; i < word.length; i++) {
			const colorIdx = (startIdx + i) % COLORS.length;
			const material = new THREE.MeshStandardMaterial({
				color: COLORS[colorIdx],
				roughness: 0.4,
				metalness: 0.2,
			});
			const mesh = createTextMesh(word[i], font, size, 0.2, material);
			mesh.position.set(-totalWidth / 2 + i * spacing, y, 0);
			scene.add(mesh);
			allLetters.push({ mesh, baseY: y, index: startIdx + i });
		}
	}

	addWord("Hello", 0.5, 0);
	addWord("Claude", -0.5, 5);

	return {
		render: (time: number) => {
			for (const l of allLetters) {
				const phase = l.index * 0.4;
				l.mesh.position.y = l.baseY + Math.sin(time * 1.5 + phase) * 0.15;
				l.mesh.rotation.y = Math.sin(time * 1.2 + phase) * 0.4;
			}
			renderer.render(scene, camera);
		},
	};
}
```

### Task 7: Create demo entry point (main.ts)

**Files:**
- Create: `src/demo/main.ts`

Initializes all demos: canvas-based in a shared animation loop, DOM-based on DOMContentLoaded.

```ts
import "ztext";
import { initGsapClean, initGsapDark, initGsapPlayful } from "./gsap-demos";
import { initTextGeoClean } from "./textgeo-clean";
import { initTextGeoDark } from "./textgeo-dark";
import { initTextGeoPlayful } from "./textgeo-playful";
import { initCleanDemo } from "./troika-clean";
import { initDarkDemo } from "./troika-dark";
import { initPlayfulDemo } from "./troika-playful";

interface Demo {
	render: (time: number) => void;
}

const demos: Demo[] = [];

function getCanvas(id: string): HTMLCanvasElement | null {
	return document.getElementById(id) as HTMLCanvasElement | null;
}

async function init() {
	// Troika demos (sync init)
	const troikaDark = getCanvas("troika-dark");
	const troikaClean = getCanvas("troika-clean");
	const troikaPlayful = getCanvas("troika-playful");

	if (troikaDark) demos.push(initDarkDemo(troikaDark));
	if (troikaClean) demos.push(initCleanDemo(troikaClean));
	if (troikaPlayful) demos.push(initPlayfulDemo(troikaPlayful));

	// TextGeometry demos (async — font load)
	const tgeoDark = getCanvas("textgeo-dark");
	const tgeoClean = getCanvas("textgeo-clean");
	const tgeoPlayful = getCanvas("textgeo-playful");

	const tgeoInits = [];
	if (tgeoDark) tgeoInits.push(initTextGeoDark(tgeoDark));
	if (tgeoClean) tgeoInits.push(initTextGeoClean(tgeoClean));
	if (tgeoPlayful) tgeoInits.push(initTextGeoPlayful(tgeoPlayful));

	const tgeoDemos = await Promise.all(tgeoInits);
	demos.push(...tgeoDemos);

	// GSAP demos (DOM-based, no render loop)
	initGsapDark();
	initGsapClean();
	initGsapPlayful();

	// ztext auto-initializes on [data-z] elements via the import

	animate();
}

function animate() {
	requestAnimationFrame(animate);
	const time = performance.now() / 1000;
	for (const demo of demos) {
		demo.render(time);
	}
}

init();
```

### Notes

- **ztext types:** The `ztext` package may not ship TypeScript declarations. If `pnpm run test` fails on the import, add `src/demo/ztext.d.ts` with `declare module "ztext";`.
- **GSAP SplitText types:** GSAP ships its own types. The `SplitText` plugin import path may need adjusting — check `gsap/SplitText` vs `gsap/dist/SplitText`.
- **Font loading:** TextGeometry demos share a cached font promise via `textgeo-shared.ts` to avoid loading the font 3 times.
- **Performance:** 6 WebGL canvases on one page. If frame rate drops, consider using IntersectionObserver to only render visible canvases. Skip for initial prototype.
